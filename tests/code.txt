; test of global scope
load 10
add 59
store A

; test of macro
macro test_macro uses x1 x2 x3
    load x1     ; loads x1 into acc
    add x2      ; adds x2 to acc
    add x3      ; adds x3 to acc

; test of subroutine
subr test_sub uses x1 x2 x3
    load x1     ; loads x1 into acc
    add x2      ; adds x2 to acc
    sub x3      ; subtracts x3 from acc
    return      ; returns back
